# vue 的自定义指令中获取元素的 offsetWidth 去优化栅格布局列表的宽高

directives:{
        setwidth(el,binding,vnode) {
            let that = vnode.context;
            if (that.itemWidth != el.offsetWidth) {
                that.itemWidth = el.offsetWidth
            }
        }
    },
